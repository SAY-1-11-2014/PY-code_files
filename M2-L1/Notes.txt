                   "Operations On List"
l = ['Sam', 'John', 'Bob', 3, 4, 5, 6.7]
print(l)
#indexing
print(l[1])
print(l[-2])
#slicing
print(l[0:4]) #[start_value : end_value +1]
#append
l.append(7.9)
print(l)
l.remove(5)
print(l)
l.pop()
print(l)
l1= [14,15,6,7,8]
l1.sort()
print(l1)
l1.reverse()
print(l1)
l1.clear()
print(l1)
